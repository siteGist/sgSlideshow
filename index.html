First we build an HTML structure, or simply the <tag/> container 

<div id="slideshow">
	<div>
	   <ul> <!-- Or in this case you can use <ol/> -->
			<li>
				The first Slide Content
			</li>
			<li>
				The second Slide Content
			</li>
			<li>
				...
			</li>
	   </ul>
	</div>
</div>
We have another option, more simple wrapping: 
<div id="slideshow">
	<li>
		The first Slide Content
	</li>
	<div> <!-- We can use div`s also ... -->
		The second Slide Content
	</div>
	<article> <!-- ... and other alowed in structure tags -->
		...
	</article>
</div>

The Options - click to expand

Slieshow behaviour

  • Automaticle go to next slide
    autoNext: true,
  • Animation modes - 'staticly' or 'scroll'
    mode: 'staticly',
  • Options for mode: 'staticly'
    staticly: {
    • Time of fadeOut (hiding) animation
      hideTime: 300,
    • Time of fadeIn (showing) animation
      showTime: 500,
    • Easing effect for fade animation
      easing: 'linear',
    • Time of first slide for fadingOut (showing)
      firstSlideShowtime: 4747
    },
  • Options for mode: 'scroll'
    scroll: {
    • Choose true if scrolling must be athwart to horisont
      vertical: false,
    • Time of animation to selected slide position
      speed: 777,
    • Easing effect for scrolling animation
      easing: 'linear',
    • Display elements in scrolling animation - 'tocurrent' or 'inline' or 'full'
      itemsLine: 'inline',
    • Carousell/circualr options
      carousell: {
      • Enable carousell mode
        enable: false,
      • This mode makes copies of original el. - 'full' or (int)Number
        copy: 1,
      • Scroll only in one direction, here copy = 'full' and you can change direction by reverse: true
        oneDirection: false
      },
    },
  • Do animation only after animation ends
    animateWhenShowEnds: false,
  • First visible slide is....
    firstSlide: 1,
  • Array, int or string. Ex.`s : [1,3,5,2,4], or 4 or 'last' or 'first'
    stopSlide: null,
  • This option is for repeating stopSlide option. true or (int) or false to repeate it once
    repeatStopSlide: true,
  • Time of slide displaying in ms
    showTime: 3000,
  • add Delay time for changing to next slide
    delay: {
    • Enable dalay time?
      enable: true,
    • Enable delay after such events: 'autoNext exactslide showprev shownext'
      enableOn: 'exactslide shownext showprev',
    • Delay time is equal to show time multiplied on this option
      multiplyTime: 3
    },
  • Show slides in reverse order. Or for scroll.oneDirectionin - in reverse direction
    reverse: false,

Markup options

  • Wrapp items in two divs via javascript?
    buildHTML: false,
  • What items must be wrapped in 2 divs (only if buildHTML: true)? Give css selector or '' for all...
    wrapSelector: '',
  • Give for each level some classes (using 1('').addClass(class) function)
    carcass: {
    • Class for main container, the root.
      container: 'container',
    • For item holder wrapper class
      wrapper: 'wrapper',
    • Class for item holder
      itemHolder: 'item-holder',
    • Class for each item
      item: 'item',
    • For curent element
      current: 'cur'
    },
  • Add size to item holder.... maybe will be usefull...
    addSizeToHolder: 0,
  • Paths to javascript libs, to load them dynamically
    jlibs: {
    • lib with easing effects
      easing: 'http://gsgd.co.uk/sandbox/jquery/easing/jquery.easing.1.3.js',
    • lib for UI animation effect - still not implemented for this plugin
      jqueryUI: 'https://ajax.googleapis.com/ajax/libs/jqueryui/1.8.13/jquery-ui.min.js'
    },
  • Options for buttons: prev/next/pages
    buttons: {
    • We are using those buttons?
      use: true,
    • Use pages buttons. Buttos for each slide.
      pages: true,
    • Use "goto next" button
      next: true,
    • Use "goto previous" button
      prev: true,
    • Hide/show buttons option - 'all', 'none', 'pages', 'next', 'prev', 'next&prev'
      hideshow: 'next&prev',
    • Bind this event to trigger button function
      actionOn: 'click',
    • More bind options
      bind: {
      • hide buttons on event
        hideOn: 'mouseleave',
      • show buttons on event
        showOn: 'mouseenter',
      • still not implemented jQueryUI animation :(
        hideStyle: 'fade',
      • still not implemented jQueryUI animation :(
        showStyle: 'fade'
      },
    • Time of buttons to hide them
      hideTime: 300,
    • Time of buttons to show them
      showTime: 500,
    • Classes for buttons (text and separate options too)
      bttnClass: {
      • Class for wrapping buttons, if separated then t wrap pages buttons
        bttnWrap: 'show-navi',
      • Class for each slide button
        pages: 'show-item',
      • Class for "goto next" button
        next: 'show-next',
      • Class for "goto previous" button
        prev: 'show-prev',
      • Text or HTML in "goto next" button
        nextText: '›',
      • Text or HTML in "goto next" button
        prevText: '‹',
      • Separate prev/next from each slide buttons?
        separate: true
      },
    },
  • Using thubs options
    thumbs: {
    • Use thumbs?
      use: false,
    • Thumbs selector in each slide item el.
      thumb: '.thumb',
    • Selector el. for changing slide view
      bindAnchor: 'a',
    • Bind this events to thumbs.bindAnchor
      bindEvent: 'click',
    • Thumbs collection position: 'left', 'right', 'top', 'bottom'
      position: 'right',
    • Height of each thumb el. - 'auto' or (int)N pixels or 'Npx'
      elHeight: 'auto',
    • Width of each thumb el. - 'auto' or (int)N pixels or 'Npx'
      elWidth: 'auto',
    • Enable scrolling for thumbs with sgSlider pugin, dont forget to include them in code
      sgSlider: false,
    • Advanced options for sgSlider
      sgSliderO: {},
    • Callback function when sgSlider created
      onSliderInit: function() {}
    },

Callback functions

  • When slide changes, n - current slide number.
    onSlideChange: function(n){},
  • On sgSlideshow starts
    onStart: function(){},
  • On start slide change animation
    onStartAnimation: function(){},
  • On end slide change animation
    onEndAnimation: function(){},
  • For slide repeating options: when auto next start launch this fn.
    onSlideRepeatStarts: function(){},
  • For slide repeating options: when auto next ends launch this fn.
    onSlideRepeatEnds: function(){}

Some samples what we used in demonstration section

$(document).ready(function() {
	$('#slideshowVScroll').sgSlideshow({
		mode: 'scroll',
		jlibs: {
		// set this only at first easing option init.
			easing: 'jquery.easing.1.3.js'
		},
		scroll: {
			easing: 'easeInQuad',
			vertical: true,
			itemsLine: 'tocurrent',
			carousell: {
				enable: true,
				oneDirection: true
			}
		}
	});
	$('#slideshowVScrollRev').sgSlideshow({
		mode: 'scroll',
		reverse: true,
		scroll: {
			easing: 'easeInExpo',
			vertical: true,
			itemsLine: 'inline',
			carousell: {
				enable: true
			}
		}
	});
	$('#slideshowHScroll').sgSlideshow({
		mode: 'scroll',
		scroll: {
			easing: 'easeInOutCirc',
			itemsLine: 'tocurrent',
			carousell: {
				enable: true,
				oneDirection: true
			}
		}
	});
	$('#slideshowHScrollRev').sgSlideshow({
		mode: 'scroll',
		reverse: true,
		stopSlide: [1,3,5,2,4],
		scroll: {
			easing: 'easeOutBounce',
			itemsLine: 'tocurrent',
			carousell: {
				copy: 'full',
				enable: true
			}
		}
	});
});
